# Chief Agent Phase 2a Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first usable LLM-driven slice of Chief Agent: real OpenRouter streaming, eight tools (Read/Glob/Grep/WebFetch/Write/Edit/Bash/AskUserQuestion), full permission + suspension framework, and a refreshed chief-chat UI. At the end, a user with an OpenRouter API key can ask Chief Agent to read files, run commands, edit code, and answer questions, with `normal`-mode permission prompts gating writes and exec.

**Architecture:** Phase 1 left clean module boundaries (`runSegment`, `getModel`, `buildSystemMessages`, `rpcCall`, `writePartUpdate`) as `Phase1NotImplemented` stubs. Phase 2a fills them in. The worker uses Vercel AI SDK v6 `streamText` to drive a segment-based loop (`runSegment` → `runLoop`), with tool wrappers that consult a worker-local pure `checkPermission` function. When permission is needed, the wrapper raises a `SuspensionSignal`, the segment ends, main shows a card, and the resolution handler (running outside any `runSegment`) actually executes the tool with its own `currentResolutionAbortController`. Local tools (Read/Glob/Grep/WebFetch/Write/Edit/AskUserQuestion) run in the worker; `Bash` is the only `rpc` tool, executed in main via `child_process.spawn`.

**Hard rules from the spec (will be re-stated where they apply, but worth memorizing):**

1. `checkPermission` returns exactly two values: `'allow' | 'prompt-needed'`. There is **no** `'deny'` return.
2. `'ui'` approval class is pass-through in **all** modes. AskUserQuestion does not go through `checkPermission`'s prompt path.
3. Tool execution for `permission-allow` happens **inside the resolution handler**, not on a subsequent re-emit.
4. **No synthetic user message** is appended for question/permission resolutions. Only plan resolutions append synthetic user messages, and Phase 2a does not implement plan mode.
5. Tool result success values are returned **raw** (no `{ok, data}` wrapper). Only error/suspension/resolved cases carry a discriminator.
6. `'__suspended'`, `'resolved'`, and `'error'` are reserved top-level keys; no tool's natural success payload may use them.
7. `currentAbortController` and `currentResolutionAbortController` are mutually exclusive; at most one is non-null at any time.

```ts
import { isAbsolute, normalize, sep } from 'node:path'
import { homedir } from 'node:os'

export class PathGuardError extends Error {
  constructor(public readonly reason: string) {
    super(`path-guard: ${reason}`)
    this.name = 'PathGuardError'
  }
}

const DENIED_HOME_DOTFILES = ['.ssh', '.aws', '.oneship', '.oneship-dev', '.claude', '.codex']

const ALLOWED_TMP_PREFIXES = ['/tmp', '/private/tmp']

export interface PathGuardOptions {
  projectRoot: string
}

/**
 * Validate that `targetPath` is safe for tool I/O.
 *
 * Rules:
 *   1. MUST be absolute.
 *   2. MUST NOT be inside any home-dir dotfile we treat as sensitive.
 *   3. MUST be under the project root OR under /tmp (or /private/tmp on macOS).
 *
 * Throws `PathGuardError` with a stable reason string on rejection.
 *
 * The path is NOT realpath-resolved by default — the function operates on
 * the literal path string. This is intentional: realpath would hide certain
 * symlink escapes, but the only legitimate case we accept (`/tmp` ↔
 * `/private/tmp`) is hardcoded into the allowed prefixes.
 */
export function guardPath(targetPath: string, opts: PathGuardOptions): void {
  if (!isAbsolute(targetPath)) {
    throw new PathGuardError('relative path not allowed')
  }

  const normalized = normalize(targetPath)
  const home = homedir()

  // Deny home dotfiles outright.
  for (const dotfile of DENIED_HOME_DOTFILES) {
    const denied = `${home}${sep}${dotfile}`
    if (normalized === denied || normalized.startsWith(denied + sep)) {
      throw new PathGuardError(`path inside ${dotfile} is denied`)
    }
  }

  // Allow if under project root.
  const projectNorm = normalize(opts.projectRoot)
  if (normalized === projectNorm || normalized.startsWith(projectNorm + sep)) {
    return
  }

  // Allow if under /tmp or /private/tmp.
  for (const prefix of ALLOWED_TMP_PREFIXES) {
    if (normalized === prefix || normalized.startsWith(prefix + sep)) {
      return
    }
  }

  throw new PathGuardError('path is outside the allowed workspace')
}
```

Create `src/agent/tools/check-permission.ts`:

```ts
import type { ApprovalClass } from '../../shared/agent-protocol'
import type { Session } from '../session/session'

export type PermissionDecision = 'allow' | 'prompt-needed'

/**
 * Pure permission evaluator. Worker-local — never crosses an IPC boundary.
 *
 * Returns exactly two values. There is NO `'deny'` return; user denials live
 * in rewritten tool_results after a resolved suspension, not in a denial cache.
 *
 * See Phase 2a §8.1 and parent §12.2.
 */
export function checkPermission(
  session: Session,
  approvalClass: ApprovalClass,
  toolName: string,
  args: unknown,
): PermissionDecision {
  // Rule 1: ui class is always pass-through, regardless of mode.
  // AskUserQuestion does its own UI mediation via the suspension protocol.
  if (approvalClass === 'ui') return 'allow'

  const mode = session.meta.permissionMode

  // Rule 2: trust mode allows everything.
  if (mode === 'trust') return 'allow'

  // Rule 3: per-class allow-always allowlist (set via Allow Always button).
  if (session.allowOnceClasses.has(approvalClass)) return 'allow'

  // Rule 4: single-use approval keys (consumed on hit).
  // Phase 2a does not populate this set in the happy path (Allow executes
  // tools during resolution), but the helper is wired for forward compat.
  const key = singleUseKey(toolName, args)
  if (session.singleUseApprovals.has(key)) {
    session.singleUseApprovals.delete(key)
    return 'allow'
  }

  // Rule 5: normal mode allows read-class tools without prompting.
  if (mode === 'normal' && approvalClass === 'read') return 'allow'

  // Rule 6: anything else needs a prompt.
  return 'prompt-needed'
}

/**
 * Stable per-call key for `singleUseApprovals`. Stable means:
 *   - same args (in any key order) → same key
 *   - different args → different key
 *
 * Implementation uses sorted-key JSON serialization. NOT cryptographically
 * stable — collision-resistant enough for in-memory session state.
 */
export function singleUseKey(toolName: string, args: unknown): string {
  return `${toolName}:${stableStringify(args)}`
}

function stableStringify(value: unknown): string {
  if (value === null || typeof value !== 'object') return JSON.stringify(value)
  if (Array.isArray(value)) return `[${value.map(stableStringify).join(',')}]`
  const keys = Object.keys(value as Record<string, unknown>).sort()
  const parts = keys.map((k) => `${JSON.stringify(k)}:${stableStringify((value as Record<string, unknown>)[k])}`)
  return `{${parts.join(',')}}`
}
```

```ts
 * Read a file and return its contents as a string. The wrapper applies

export async function readTool(args: ReadArgs): Promise<string> {
  const content = await fs.readFile(args.file_path, 'utf-8')
  if (args.offset === undefined && args.limit === undefined) {

  }
  const lines = content.split('\n')

  const end = args.limit !== undefined ? start + args.limit : lines.length
  return lines.slice(start, end).join('\n')
}
```

```ts
import fg from 'fast-glob'
import { promises as fs } from 'node:fs'

export interface GrepArgs {
  pattern: string
  path?: string
  glob?: string
  case_insensitive?: boolean
}

export interface GrepMatch {
  file: string
  line: number
  text: string
}

/**
 * Search file contents using a regular expression. Phase 2a uses Node's
 * RegExp (no ripgrep dependency). For each matching line, returns
 * { file, line (1-indexed), text }.
 */
export async function grepTool(args: GrepArgs): Promise<GrepMatch[]> {
  const cwd = args.path ?? process.cwd()
  const filePattern = args.glob ?? '**/*'
  const files = await fg(filePattern, {
    cwd,
    absolute: true,
    dot: false,
    onlyFiles: true,
    ignore: ['**/node_modules/**', '**/.git/**'],
  })

  const flags = args.case_insensitive ? 'i' : ''
  const re = new RegExp(args.pattern, flags)
  const matches: GrepMatch[] = []

  for (const file of files) {
    let content: string
    try {
      content = await fs.readFile(file, 'utf-8')
    } catch {
      continue // skip binaries / unreadable
    }
    const lines = content.split('\n')
    for (let i = 0; i < lines.length; i++) {
      if (re.test(lines[i])) {
        matches.push({ file, line: i + 1, text: lines[i] })
      }
    }
  }

  return matches
}
```

```ts
export async function webFetchTool(args: WebFetchArgs): Promise<WebFetchResult> {

  try {

    return { url: args.url, status: res.status, body: truncated }
  } finally {
    clearTimeout(timeout)
  }
}
```

```ts
import { promises as fs } from 'node:fs'

export interface EditArgs {
  file_path: string
  old_string: string
  new_string: string
  replace_all?: boolean
}

export interface EditResult {
  file_path: string
  replacements: number
}

/**
 * Replace exact string occurrences in a file. Throws if old_string is not
 * found. Default replaces only the first occurrence; replace_all=true does
 * all matches.
 */
export async function editTool(args: EditArgs): Promise<EditResult> {
  const content = await fs.readFile(args.file_path, 'utf-8')
  if (!content.includes(args.old_string)) {
    throw new Error(`old_string not found in ${args.file_path}`)
  }
  let updated: string
  let count: number
  if (args.replace_all) {
    updated = content.split(args.old_string).join(args.new_string)
    count = content.split(args.old_string).length - 1
  } else {
    const idx = content.indexOf(args.old_string)
    updated = content.slice(0, idx) + args.new_string + content.slice(idx + args.old_string.length)
    count = 1
  }
  await fs.writeFile(args.file_path, updated, 'utf-8')
  return { file_path: args.file_path, replacements: count }
}
```

```ts
import { tool, type Tool } from 'ai'
import { nanoid } from 'nanoid'
import type { ToolManifest } from '../../shared/tool-manifest'
import { checkPermission } from './check-permission'
import { SuspensionSignal } from './suspension-signal'
import { guardPath, PathGuardError } from '../../shared/tool-guards'
import type { Session } from '../session/session'

export interface ExecContext {
  toolCallId: string
  abortSignal: AbortSignal
  experimental_context: {
    session: Session
    projectRoot: string
  }
}

const PATH_AWARE_TOOLS = new Set(['Read', 'Write', 'Edit'])

/**
 * Wrap a manifest entry + inner executor function as a Vercel AI SDK
 * `tool()`. The wrapper applies permission check, guard checks, and
 * suspension raising in front of the inner executor.
 *
 * For `ui` class tools (AskUserQuestion), `inner` is never called — the
 * wrapper raises a `kind: 'question'` SuspensionSpec instead.
 *
 * See Phase 2a §6.2 and §8.2.
 */
export function wrapLocalTool<TArgs, TResult>(
  manifest: ToolManifest,
  inner: (args: TArgs, ctx: ExecContext) => Promise<TResult>,
): Tool {
  return tool({
    description: manifest.description,
    inputSchema: manifest.inputSchema as any,
    execute: async (args: any, ctx: any) => {
      const session: Session = ctx.experimental_context.session
      const projectRoot: string = ctx.experimental_context.projectRoot
      const toolCallId: string = ctx.toolCallId

      // Special path: ui class is a control-flow tool. Build the question
      // SuspensionSpec directly and throw, without calling inner.
      if (manifest.approvalClass === 'ui') {
        return raiseQuestionSuspension(session, manifest, args, toolCallId)
      }

      const decision = checkPermission(session, manifest.approvalClass, manifest.name, args)

      if (decision === 'prompt-needed') {
        return raisePermissionSuspension(session, manifest, args, toolCallId)
      }

      // decision === 'allow'
      try {
        if (PATH_AWARE_TOOLS.has(manifest.name)) {
          guardPath((args as { file_path: string }).file_path, { projectRoot })
        }
        return await inner(args, ctx)
      } catch (err) {
        if (err instanceof PathGuardError) {
          return { error: `path-guard: ${err.reason}` }
        }
        if (err instanceof SuspensionSignal) {
          throw err // propagate; not an error
        }
        const message = err instanceof Error ? err.message : String(err)
        return { error: message }
      }
    },
  })
}

function raisePermissionSuspension(
  session: Session,
  manifest: ToolManifest,
  args: unknown,
  toolCallId: string,
): never {
  const suspensionId = nanoid()
  session.pendingSuspension = {
    kind: 'permission',
    suspensionId,
    messageId: session.currentMessageId,
    partIndex: session.currentPartIndex + 1,
    toolCallId,
    toolName: manifest.name,
    approvalClass: manifest.approvalClass,
    summary: manifest.summarize(args),
    args,
  }
  // The placeholder tool-result is written by Session.appendPlaceholderToolResult
  // BEFORE the SuspensionSignal is thrown, so replay sees the placeholder
  // even on a crash mid-suspension. The actual write call lives on Session
  // (Task 10) — for now, callers must invoke it.
  // NOTE for Task 10: refactor so the wrapper calls
  // session.appendPlaceholderToolResult({ messageId, toolCallId, suspensionId })
  // synchronously before throwing.
  void session.appendPlaceholderToolResult?.({
    messageId: session.currentMessageId,
    toolCallId,
    result: { __suspended: true, suspensionId },
  })
  throw new SuspensionSignal(suspensionId)
}

function raiseQuestionSuspension(
  session: Session,
  manifest: ToolManifest,
  args: unknown,
  toolCallId: string,
): never {
  const suspensionId = nanoid()
  const a = args as { question: string; choices?: Array<{ id: string; label: string; description?: string }> }
  session.pendingSuspension = {
    kind: 'question',
    suspensionId,
    messageId: session.currentMessageId,
    partIndex: session.currentPartIndex + 1,
    toolCallId,
    question: a.question,
    choices: a.choices,
  }
  void session.appendPlaceholderToolResult?.({
    messageId: session.currentMessageId,
    toolCallId,
    result: { __suspended: true, suspensionId },
  })
  throw new SuspensionSignal(suspensionId)
}
```

```ts
import type { Session } from './session'
import type { SuspensionResolution, ApprovalClass } from '../../shared/agent-protocol'
import { TOOL_MANIFESTS } from '../../shared/tool-manifest'
import { readTool } from '../tools/read'
import { writeTool } from '../tools/write'
import { editTool } from '../tools/edit'
import { globTool } from '../tools/glob'
import { grepTool } from '../tools/grep'
import { webFetchTool } from '../tools/web-fetch'
import { guardPath, PathGuardError } from '../../shared/tool-guards'
import { existsSync, unlinkSync } from 'node:fs'

export interface StateUpdate {
  addAllowOnceClass?: ApprovalClass
}

export interface ResolutionContext {
  projectRoot: string
}

const LOCAL_TOOL_DISPATCH: Record<string, (args: any) => Promise<unknown>> = {
  Read: readTool,
  Write: writeTool,
  Edit: editTool,
  Glob: globTool,
  Grep: grepTool,
  WebFetch: webFetchTool,
}

const PATH_AWARE_TOOLS = new Set(['Read', 'Write', 'Edit'])

/**
 * Apply a SuspensionResolution to a session that has a pendingSuspension.
 *
 * Per Phase 2a §8.2 step 10:
 *   - Apply stateUpdate (if any) to session.allowOnceClasses
 *   - For permission-allow / permission-allow-always: invoke the real tool
 *     now, capture its output, rewrite the placeholder
 *   - For permission-deny: rewrite to {error: 'user-denied'}, no execution
 *   - For question-answered: rewrite to {resolved: true, answer}
 *
 * Sets up session.currentResolutionAbortController for the duration of the
 * tool execution (so cancel-current-turn can abort it). Clears the field in
 * finally. Per the §13.3 invariant, currentAbortController is null at this
 * point.
 *
 * Does NOT trigger a new runLoop — the caller is responsible for that.
 */
export async function handleResolveSuspension(
  session: Session,
  resolution: SuspensionResolution,
  stateUpdate: StateUpdate | undefined,
  ctx: ResolutionContext,
): Promise<void> {
  const spec = session.pendingSuspension
  if (!spec) {
    throw new Error('handleResolveSuspension: no pending suspension')
  }

  // Apply state update first.
  if (stateUpdate?.addAllowOnceClass) {
    session.allowOnceClasses.add(stateUpdate.addAllowOnceClass)
  }

  let resolvedResult: unknown

  if (resolution.kind === 'permission-deny') {
    resolvedResult = { error: 'user-denied' }
  } else if (resolution.kind === 'permission-allow' || resolution.kind === 'permission-allow-always') {
    if (spec.kind !== 'permission') {
      throw new Error('permission-allow resolution requires kind: permission')
    }
    resolvedResult = await executeDeferredTool(session, spec.toolName, spec.args, ctx)
  } else if (resolution.kind === 'question-answered') {
    resolvedResult = { resolved: true, answer: resolution.answer }
  } else if (resolution.kind === 'plan-approved' || resolution.kind === 'plan-modified' || resolution.kind === 'plan-rejected') {
    throw new Error(`Phase2bNotImplemented: plan resolution ${resolution.kind}`)
  } else {
    const _exhaustive: never = resolution
    throw new Error(`unknown resolution kind: ${_exhaustive}`)
  }

  await session.rewriteToolResultPart({
    messageId: spec.messageId,
    partIndex: spec.partIndex,
    result: resolvedResult,
  })

  await deleteSuspensionFile(session)
  session.pendingSuspension = null
}

async function executeDeferredTool(
  session: Session,
  toolName: string,
  args: unknown,
  ctx: ResolutionContext,
): Promise<unknown> {
  const manifest = TOOL_MANIFESTS[toolName]
  if (!manifest) return { error: `unknown tool: ${toolName}` }

  if (manifest.execution !== 'local') {
    // Phase 2a: only Bash is rpc, and it goes through main. The main-side
    // resolution handler will handle Bash; this branch should not be hit
    // for local-tool executions.
    return { error: `executeDeferredTool: rpc tools (${toolName}) must be invoked via main` }
  }

  const inner = LOCAL_TOOL_DISPATCH[toolName]
  if (!inner) return { error: `no executor for ${toolName}` }

          return { error: `path-guard: ${err.reason}` }

    return await inner(args)
  } catch (err) {
    if ((err as Error).name === 'AbortError') {
      return { error: 'user-cancelled', hint: 'Cancelled during resolution' }
    }
    return { error: err instanceof Error ? err.message : String(err) }

    session.currentResolutionAbortController = null
```

```ts
const FLUSH_WINDOW_MS = 50

/**
 * Buffers streaming text deltas and flushes them as accumulated snapshots.
 *
 * Per Phase 2a §7.3 and parent §15.2.3:
 *   - append(delta) accumulates into a string buffer
 *   - if no timer is running, starts a 50ms setTimeout
 *   - flush() writes the FULL accumulated string (NOT a delta) via the
 *     `onFlush` callback — this is because the spec's `part-update` event
 *     is a full-part replacement, not a delta patch
 *   - the buffer is NOT cleared after a flush — subsequent appends grow
 *     the same string, so the next flush carries the full snapshot
 *   - finalFlush() forces an immediate flush and is idempotent
 */
export class TextFlushScheduler {
  private buffer = ''
  private hasUnflushedAppends = false
  private timer: NodeJS.Timeout | null = null
  private flushInFlight: Promise<void> | null = null

  constructor(private readonly onFlush: (accumulated: string) => Promise<void>) {}

  append(delta: string): void {
    this.buffer += delta
    this.hasUnflushedAppends = true
    if (!this.timer) {
      this.timer = setTimeout(() => {
        void this.flushInternal()
      }, FLUSH_WINDOW_MS)
    }
  }

  async finalFlush(): Promise<void> {
    if (this.timer) {
      clearTimeout(this.timer)
      this.timer = null
    }
    if (this.hasUnflushedAppends) {
      await this.flushInternal()
    }
  }

  private async flushInternal(): Promise<void> {
    if (this.flushInFlight) {
      await this.flushInFlight
      return
    }
    this.timer = null
    this.hasUnflushedAppends = false
    const snapshot = this.buffer
    this.flushInFlight = this.onFlush(snapshot)
    try {
      await this.flushInFlight
    } finally {
      this.flushInFlight = null
    }
  }
}
```

```ts
export type ErrorClass = 'retryable' | 'non-retryable' | 'abort'

export interface RetryOptions {
  backoffMs?: number[]
}

const DEFAULT_BACKOFF_MS = [200, 800, 3200]

/**
 * Classify an error for retry decisions.
 *
 * Retryable: HTTP 429/529, network errors (ECONNRESET, ETIMEDOUT, EPIPE).
 * Non-retryable: HTTP 4xx (other than 429), HTTP 5xx (other than 529), unknown.
 * Abort: AbortError — caller treats as "user cancelled".
 */
export function classifyError(err: any): ErrorClass {
  if (err?.name === 'AbortError') return 'abort'
  if (typeof err?.status === 'number') {
    if (err.status === 429 || err.status === 529) return 'retryable'
    return 'non-retryable'
  }
  if (typeof err?.code === 'string') {
    if (['ECONNRESET', 'ETIMEDOUT', 'EPIPE'].includes(err.code)) return 'retryable'
  }
  return 'non-retryable'
}

/**
 * Retry an async fn on retryable errors with exponential backoff.
 * Default backoff: 200ms, 800ms, 3200ms (3 retries → 4 total attempts).
 *
 * Throws abort errors immediately (no retry, no wrapping). Throws the LAST
 * retryable error after exhausting all attempts.
 */
export async function withRetry<T>(
  fn: () => Promise<T>,
  opts: RetryOptions = {},
): Promise<T> {
  const backoff = opts.backoffMs ?? DEFAULT_BACKOFF_MS
  const maxAttempts = backoff.length + 1
  let lastError: unknown
  for (let attempt = 0; attempt < maxAttempts; attempt++) {
    try {
      return await fn()
    } catch (err) {
      const klass = classifyError(err)
      if (klass === 'abort') throw err
      if (klass === 'non-retryable') throw err
      lastError = err
      if (attempt < backoff.length) {
        await sleep(backoff[attempt])
      }
    }
  }
  throw lastError
}

function sleep(ms: number): Promise<void> {
  return new Promise((resolve) => setTimeout(resolve, ms))
}
```

```ts
const STEP_CAP = 50

/**
 * Run one segment of the chief agent loop. See Phase 2a §7.1.
 *
 * Returns when:
 *   - the LLM finishes naturally → 'natural'
 *   - the LLM finishes via tool-calls → 'tool-calls' (runLoop continues)
 *   - the step cap fires → 'step-cap'
 *   - a tool throws SuspensionSignal → 'suspended'
 *   - externalAbort fires → 'aborted'
 *   - any other error → 'error'
 */
export async function runSegment(
  session: Session,
  externalAbort: AbortSignal,
  deps: RunSegmentDeps = {},
): Promise<RunSegmentResult> {
  const stream = deps.streamText ?? streamText
  const model = (deps.getModel ?? getModel)(session.meta.model).model

  // Begin a new assistant message for this segment.
  await session.beginAssistantMessage()
  const messageId = session.currentMessageId!
  let textPartIndex = -1
  const flusher = new TextFlushScheduler(async (accumulated) => {
    if (textPartIndex < 0) {
      // First flush: append a new text part.
      await session.appendTextPart({ text: accumulated })
      textPartIndex = session.currentPartIndex
    } else {
      await session.rewriteTextPart({ messageId, partIndex: textPartIndex, text: accumulated })
    }
  })

  let stepCount = 0
  let result: RunSegmentResult

  try {
    const llmResult = await withRetry(async () => {
      return stream({
        model: model as LanguageModel,
        messages: buildSystemMessages(session).concat(session.toCoreMessages()),
        tools: buildTools(),
        stopWhen: stepCountIs(STEP_CAP),
        abortSignal: externalAbort,
        experimental_context: { session, projectRoot: session.projectRoot ?? process.cwd() },
      })
    })

    for await (const part of llmResult.fullStream) {
      if (part.type === 'text-delta') {
        flusher.append(part.textDelta)
      } else if (part.type === 'tool-call') {
        await flusher.finalFlush()
        textPartIndex = -1   // next text-delta starts a new text part
        await session.appendToolCallPart({
          toolCallId: part.toolCallId,
          toolName: part.toolName,
          toolArgs: part.input,
        })
        stepCount++
      } else if (part.type === 'tool-result') {
        await session.appendPlaceholderToolResult({
          messageId,
          toolCallId: part.toolCallId,
          result: part.output,
        })
      } else if (part.type === 'finish') {
        await flusher.finalFlush()
        await session.endAssistantMessage()
        result = mapFinishReason(part.finishReason, stepCount)
        return result
      }
    }
    // Stream ended without a finish event — treat as natural.
    await flusher.finalFlush()
    await session.endAssistantMessage()
    return { reason: 'natural' }
  } catch (err) {
    await flusher.finalFlush().catch(() => {})
    if (err instanceof SuspensionSignal) {
      // pendingSuspension is already set by the wrapper; persist it.
      await session.persistSuspension()
      await session.endAssistantMessage()
      return { reason: 'suspended' }
    }
    if ((err as Error).name === 'AbortError') {
      await session.endAssistantMessage()
      return { reason: 'aborted' }
    }
    await session.endAssistantMessage()
    return { reason: 'error', error: sanitizeError(err) }
  }

  if (finishReason === 'content-filter') return { reason: 'error', error: 'content-filter' }
  return { reason: 'error', error: `unknown-finish-reason: ${finishReason}` }
}
```

```ts
import { existsSync, readFileSync, writeFileSync, chmodSync, mkdirSync } from 'node:fs'
import { dirname, join } from 'node:path'
import { runtimePaths } from './runtime-paths'

interface SecretFile {
  secrets: Record<string, string>
}

/**
 * Phase 2a secret-store backend: a JSON file at
 * `<globalState>/secrets.json` with permissions 0600.
 *
 * V1 swaps this for keytar without changing the external API.
 */

function secretsPath(): string {
  return join(runtimePaths().globalState, 'secrets.json')
}

function load(): SecretFile {
  const path = secretsPath()
  if (!existsSync(path)) return { secrets: {} }
  try {
    return JSON.parse(readFileSync(path, 'utf-8'))
  } catch {
    return { secrets: {} }
  }
}

function save(data: SecretFile): void {
  const path = secretsPath()
  mkdirSync(dirname(path), { recursive: true })
  writeFileSync(path, JSON.stringify(data, null, 2), 'utf-8')
  chmodSync(path, 0o600)
}

export function getSecret(name: string): string | undefined {
  return load().secrets[name]
}

export function setSecret(name: string, value: string): void {
  const data = load()
  data.secrets[name] = value
  save(data)
}

export function deleteSecret(name: string): void {
  const data = load()
  delete data.secrets[name]
  save(data)
}
```

```ts
import { spawn } from 'node:child_process'
import type { Readable } from 'node:stream'

export interface BashArgs {
  cmd: string
  cwd?: string
}

export interface BashResult {
  stdout: string
  stderr: string
  exitCode: number | null
  truncated?: boolean
}

const MAX_BUFFER_BYTES = 1_000_000  // 1 MB per stream

/**
 * Run a one-shot bash command. Captures stdout/stderr (capped at 1 MB each)
 * and the exit code. Honors AbortSignal — sends SIGTERM then SIGKILL after
 * 200ms.
 *
 * Phase 2a does NOT use TerminalManager. See Phase 2a §11.4.
 */
export async function bashExecute(args: BashArgs, signal: AbortSignal): Promise<BashResult> {
  return new Promise((resolve, reject) => {
    const child = spawn('bash', ['-c', args.cmd], {
      cwd: args.cwd ?? process.cwd(),
      env: process.env,
      stdio: ['ignore', 'pipe', 'pipe'],
    })

    let stdoutBuf = ''
    let stderrBuf = ''
    let stdoutTruncated = false
    let stderrTruncated = false

    const accumulate = (which: 'out' | 'err') => (chunk: Buffer | string) => {
      const text = typeof chunk === 'string' ? chunk : chunk.toString('utf-8')
      if (which === 'out') {
        if (stdoutBuf.length + text.length > MAX_BUFFER_BYTES) {
          stdoutTruncated = true
          stdoutBuf = stdoutBuf.slice(0, MAX_BUFFER_BYTES / 2) + '\n…[truncated]…\n' + text.slice(-MAX_BUFFER_BYTES / 2)
        } else {
          stdoutBuf += text
        }
      } else {
        if (stderrBuf.length + text.length > MAX_BUFFER_BYTES) {
          stderrTruncated = true
          stderrBuf = stderrBuf.slice(0, MAX_BUFFER_BYTES / 2) + '\n…[truncated]…\n' + text.slice(-MAX_BUFFER_BYTES / 2)
        } else {
          stderrBuf += text
        }
      }
    }

    child.stdout?.on('data', accumulate('out'))
    child.stderr?.on('data', accumulate('err'))

    const onAbort = () => {
      child.kill('SIGTERM')
      setTimeout(() => {
        if (!child.killed) child.kill('SIGKILL')
      }, 200)
    }
    signal.addEventListener('abort', onAbort, { once: true })

    child.on('error', reject)
    child.on('close', (code) => {
      signal.removeEventListener('abort', onAbort)
      const result: BashResult = {
        stdout: stdoutBuf,
        stderr: stderrBuf,
        exitCode: code,
      }
      if (stdoutTruncated || stderrTruncated) result.truncated = true
      resolve(result)
    })
  })
}
```

```ts
import type { BrowserWindow } from 'electron'
import type { SuspensionSpec } from '../shared/agent-protocol'
import { nanoid } from 'nanoid'

interface PendingCard {
  cardId: string
  sessionId: string
  spec: SuspensionSpec
}

const pendingCards = new Map<string, PendingCard>()

/**
 * Receive a suspension-raised message from worker, generate a card id,
 * and dispatch the appropriate prompt to the renderer.
 */
export function handleSuspensionRaised(
  win: BrowserWindow,
  sessionId: string,
  spec: SuspensionSpec,
): string {
  const cardId = nanoid()
  pendingCards.set(cardId, { cardId, sessionId, spec })

  if (spec.kind === 'permission') {
    win.webContents.send('chief:event', {
      type: 'permission.prompt',
      cardId,
      spec,
    })
  } else if (spec.kind === 'question') {
    win.webContents.send('chief:event', {
      type: 'ask.prompt',
      cardId,
      question: spec.question,
      choices: spec.choices,
    })
  } else if (spec.kind === 'plan-proposal') {
    throw new Error('Phase2bNotImplemented: plan-proposal')
  }

  return cardId
}

export function getCard(cardId: string): PendingCard | undefined {
  return pendingCards.get(cardId)
}

export function clearCard(cardId: string): void {
  pendingCards.delete(cardId)
}

export function cancelAllCardsForSession(win: BrowserWindow, sessionId: string): void {
  for (const [cardId, card] of pendingCards) {
    if (card.sessionId === sessionId) {
      pendingCards.delete(cardId)
      const channel = card.spec.kind === 'permission' ? 'permission.cancel' : 'ask.cancel'
      win.webContents.send('chief:event', { type: channel, cardId })
    }
  }
}
```

```ts
    }
    expect(toolCallSeen).toBe(true)
  }, 60_000)
```
